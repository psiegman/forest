 54 /**
 55  * Calculates the distance between the two points in meters using the Haversine method.
 56  * source: http://www.movable-type.co.uk/scripts/latlong.html
 57  */
 58 GeoPosition.prototype.calculateDistance = function (lat1, lon1, lat2, lon2) {
 59     var R = 6371000; // earth radius in meters
 60     var dLat = this.toRad(lat2 - lat1);
 61     var dLon = this.toRad(lon2 - lon1);
 62     var a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
 63         Math.cos(this.toRad(lat1)) * Math.cos(this.toRad(lat2)) *
 64         Math.sin(dLon / 2) * Math.sin(dLon / 2);
 65     var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
 66     var d = R * c;
 67     return d;
 68 }
 69 
 70 GeoPosition.prototype.toRad = function (x) {
 71     return x * Math.PI / 180;
 72 }

 90 /**
 91  * Calculates the distances between the two gps points.
 92  * @return {distance: total distance traveled in meters,
 93  *  distanceX: distance traveled east-west/X-axis in meters
 94  *  distanceY: distance traveled north-sout/Y-axis in meters
 95  * }
 96  * Distance x/y will be negative if traveled south/west.
 97  */
 98 GeoPosition.prototype.calculateDistances = function(lat1, lon1, lat2, lon2) {
 99     var distance = this.calculateDistance(lat1, lon1, lat2, lon2);
100 
101     // distance traveled over latitude/y-axis in meters
102     // calculated by measuring the distance between old and new lat, keeping lon constant
103     var distanceY = this.calculateDistance(lat1, lon1, lat2, lon1); // yes, lon1 twice
104 
105     // distance traveled over longitude/x-axis in meters
106     // calculated by pythagoras formula on distance and distanceY
107     var distanceX = Math.sqrt((distance * distance) - (distanceY * distanceY));
108 
109     distanceX = this.calculateXSign(lon1, lon2) * distanceX;
110     distanceY = this.calculateYSign(lat1, lat2) * distanceY;
111 
112     return {
113         distance: distance,
114         distanceX: distanceX,
115         distanceY: distanceY
116     };
117 }

124 /**
125  * Update current speed/distance/heading with information from the given geoLocation.
126  */
127 GeoPosition.prototype.updateGeoLocation = function(geoLocation) {
128     var pos = geoLocation.coords;
129     console.log("latitude:" + pos.latitude + ", longitude:" + pos.longitude + ", accuracy: " + pos.accuracy + ", heading: " + pos.heading + ", speed:" + pos.speed);
130     if (this.currentGeoLocation === undefined) {
131         this.speed = 0;
132         this.currentGeoLocation = geoLocation;
133 
134         if (this.zeroLat === undefined) {
135             this.zeroLat = this.currentGeoLocation.coords.latitude;
136             this.zeroLon = this.currentGeoLocation.coords.longitude;
137             var absoluteDistances = this.calculateDistances(this.zeroLat, this.zeroLon, geoLocation.coords.latitude, geoLocation.coords.longitude);
138             this.positionX = absoluteDistances.distanceX;
139             this.positionY = absoluteDistances.distanceY;
140         } else {
141             this.positionX = 0;
142             this.positionY = 0;
143         }
144 
145         return {
146             x: this.positionX,
147             y: this.positionY
148         }
149     }
150 
151     var currentPos = this.currentGeoLocation.coords;
152     var newPos = geoLocation.coords;
153 
154     var distances = this.calculateDistances(
155         currentPos.latitude, 
156         currentPos.longitude, 
157         newPos.latitude, 
158         newPos.longitude);
159     var dTime = geoLocation.timestamp - this.currentGeoLocation.timestamp;
160     this.speed = 1000 * (distances.distance / dTime);
161     this.speedX = 1000 * (distances.distanceX / dTime);
162     this.speedY = 1000 * (distances.distanceY / dTime);
163 
164     this.currentGeoLocation = geoLocation;
165 
166     var absoluteDistances = this.calculateDistances(this.zeroLat, this.zeroLon, newPos.latitude, newPos.longitude);
167     this.positionX = absoluteDistances.distanceX;
168     this.positionY = absoluteDistances.distanceY;
169 
170     return {
171         x: this.positionX,
172         y: this.positionY
173     }
174 }

176 /**
177  * Whether the distance traveled was in the eastern direction (returning 1) or west (returning -1)
178  * Will not work correctly if oldLon,newLon are crossing the 180th meridian
179  */
180 GeoPosition.prototype.calculateXSign = function (oldLon, newLon) {
181     if (oldLon > newLon) {
182         return -1;
183     } else {
184         return 1;
185     }
186 }
187 
188 /**
189  * Whether the distance traveled was in the northern direction (returning 1) or south (returning -1)
190  * Will not work correctly if oldLat and newLat are on opposite sides of the north or sout pole.
191  */
192 GeoPosition.prototype.calculateYSign = function (oldLat, newLat) {
193     if (oldLat > newLat) {
194         return -1;
195     } else {
196         return 1;
197     }
198 }

208 /**
209  * Assuming we move at a constant speed and rate then where will be at the given time
210  * @param time time in milliseconds since January 1st, 1970
211  */
212 GeoPosition.prototype.getPositionAtTime = function (time) {
213     if (this.currentGeoLocation === undefined) {
214         return undefined;
215     }
216 
217     if (this.speed == 0) {
218         return {
219             x: this.positionX,
220             y: this.positionY
221         }
222     }
223 
224 
225     var dTime = (time - this.currentGeoLocation.timestamp) / 1000;
226     var timedPositionX = this.positionX + (this.speedX * dTime);
227     var timedPositionY = this.positionY + (this.speedY * dTime);
228     return {
229         x: timedPositionX,
230         y: timedPositionY
231     }
232 }

234 /**
235  * Calculates where we end up if we travel from point a to point b for distance d.
236  *
237  * @return {{positionX: the position on the x-axis, positionY: the position on the y-axis}}
238  */
239 GeoPosition.prototype.getPositionAtDistance = function (x1, y1, x2, y2, d) {
240 
241     var angle = Math.abs(y2 - y1) / Math.abs(x2 - x1);
242 
243     var dX = (d * Math.cos(angle));
244     if (x1 > x2) {
245         dX = -dX;
246     }
247     var x = x1 + dX;
248 
249     var dY = (d * Math.sin(angle));
250     if (y1 > y2) {
251         dY = -dY;
252     }
253     var y = y1 + dY;
254 
255     return {
256         x: x,
257         y: y
258     }
259 }

262 /**
263  * The position at dTime in the future if we move from our current position to where we should be according to the GPS.
264  *
265  * @param currentX our position in meters relative to zeroLat
266  * @param currentY our position in meters relative to zerLon
267  * @param dTime time in milliseconds from now
268  * @param currentTime (optional) the current time in milliseconds since January 1st, 1970. Useful for testing purposes, defaults to the actual current time.
269  *
270  * @return {positionX: our position-to-be in meters relative to zerLat, positionY: our position-to-be in meters relative to zerLon, speed: our current speed in meters per second, lastGeoLocation: the last geoLocation used to calculate all of this}
271  */
272 GeoPosition.prototype.getFuturePosition = function (currentX, currentY, dTime, currentTime) {
273     // GPS is not initialized yet
274     if (this.currentGeoLocation === undefined) {
275         return {
276             x: currentX,
277             y: currentY
278         }
279     }
280 
281     // we're standing still
282     if (this.speed === 0) {
283         return {
284             x: currentX,
285             y: currentY,
286             speed: 0,
287             lastGeoLocation: this.currentGeoLocation
288         }
289     }
290 
291     // where should we be at time timestamp
292     currentTime = currentTime || this.getCurrentTimestamp();
293     var whereWeShouldBeAtDTime = this.getPositionAtTime(currentTime + dTime);
294 
295     // travel at our current speed during dTime from our current postion in the direction of where we should be
296     var distance = this.speed * (dTime / 1000);
297 
298     var positionAtDistance = this.getPositionAtDistance(currentX, currentY, whereWeShouldBeAtDTime.x, whereWeShouldBeAtDTime.y, distance);
299 
300     return {
301         x: positionAtDistance.x,
302         y: positionAtDistance.y,
303         speed: this.speed,
304         lastGeoLocation: this.currentGeoLocation
305     };
306 }
